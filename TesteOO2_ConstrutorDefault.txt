{
	classe B{
		
		int y = 0,
		int soma = 0;
		
		proc set(int v){
			this.y := v
		},
		
		proc somar(int v){
			write(this.y + v)
		}
	},
	
	classe A extends B{
		
		int x = 0;
		
		A(int v){
			this.x := v
		},

		proc print(){
			write(this.soma)
		}
		
	};
	
	{
		A a := new A(3);
		a.set(2);
		a.somar(a.x)
	}

}
